```python
def _portfolio_weighted_aggregate(
    detail_df: pd.DataFrame,
    investable_benchmarks: list[str],
    base_weights: dict[str, float],
) -> float | None:
    """Compute portfolio-weight-aligned aggregate P(Actionable Sell).

    Filters detail_df to investable benchmarks with valid calibrated
    probabilities, renormalizes base_weights over the available subset,
    and returns a weighted-average probability.

    Returns None if no investable benchmarks are available.
    """
    mask = (
        detail_df["benchmark"].isin(investable_benchmarks)
        & detail_df["classifier_prob_actionable_sell"].notna()
    )
    available = detail_df.loc[mask].copy()
    if available.empty:
        return None

    available["_base_weight"] = available["benchmark"].map(base_weights).fillna(0.0)
    total_weight = available["_base_weight"].sum()
    if total_weight <= 0.0:
        return None

    available["_norm_weight"] = available["_base_weight"] / total_weight
    aggregate = float(
        (available["classifier_prob_actionable_sell"] * available["_norm_weight"]).sum()
    )
    return aggregate
```

# ---------------------------------------------------------------------------
# Portfolio composite return builder

def build_composite_return_series(
    conn: sqlite3.Connection,
    pgr_benchmark: str = "VOO",  # PGR relative return uses PGR price series implicitly
    weights: dict[str, float] | None = None,
    as_of: date | None = None,
) -> pd.Series:
    """Build composite portfolio relative return: r_PGR - Σ(w_i * r_i).

    Loads the 6-month relative return for each investable benchmark from the
    DB (same pipeline used by per-benchmark classifiers), then computes a
    weighted-average benchmark return subtracted from PGR's return implicitly
    encoded in the per-benchmark relative returns.

    Because each relative return series r_i = r_PGR - r_benchmark_i, we have:
        r_PGR - Σ(w_i * r_i_benchmark) = Σ(w_i * (r_PGR - r_benchmark_i))
                                         = Σ(w_i * rel_return_i)

    So the composite relative return is simply Σ(w_i * rel_return_i).
    """
    if weights is None:
        weights = COMPOSITE_WEIGHTS

    rel_series_list: list[pd.Series] = []
    weight_list: list[float] = []

    for ticker, weight in weights.items():
        try:
            rel = load_relative_return_matrix(conn, ticker, HORIZON_MONTHS)
            if rel is not None and not rel.empty:
                if as_of is not None:
                    rel = truncate_relative_target_for_asof(
                        rel, as_of=pd.Timestamp(as_of), horizon_months=HORIZON_MONTHS
                    )
                rel_series_list.append(rel.rename(ticker))
                weight_list.append(weight)
        except Exception:
            continue

    if not rel_series_list:
        raise ValueError("No relative return series could be loaded for composite target.")

    # Renormalize weights over available tickers
    total_w = sum(weight_list)
    norm_weights = [w / total_w for w in weight_list]

    # Align on common dates and compute weighted composite
    combined = pd.concat(rel_series_list, axis=1).dropna(how="all")
    composite = sum(
        combined[s.name] * w
        for s, w in zip(rel_series_list, norm_weights)
        if s.name in combined.columns
    )
    composite.name = "composite_relative_return"
    return composite.dropna()

def _compute_ece(y_true: np.ndarray, y_prob: np.ndarray, n_bins: int = 10) -> float:
    """Expected Calibration Error (equal-width bins)."""
    bins = np.linspace(0.0, 1.0, n_bins + 1)
    ece = 0.0
    for lo, hi in zip(bins[:-1], bins[1:]):
        mask = (y_prob >= lo) & (y_prob < hi)
        if mask.sum() == 0:
            continue
        bin_acc = float(y_true[mask].mean())
        bin_conf = float(y_prob[mask].mean())
        ece += mask.sum() * abs(bin_acc - bin_conf)
    return ece / max(len(y_true), 1)


def run_wfo(
    X: pd.DataFrame,
    y: pd.Series,
    n_splits: int = N_SPLITS,
    test_size: int = TEST_SIZE,
    gap: int = GAP,
    min_train_obs: int = MIN_TRAIN_OBS,
) -> pd.DataFrame:
    """Walk-forward OOS evaluation with TimeSeriesSplit + gap.

    Returns a DataFrame with columns:
        fold, train_start, train_end, test_start, test_end,
        y_true, y_prob, fold_balanced_acc, fold_brier
    """
    tscv = TimeSeriesSplit(n_splits=n_splits, test_size=test_size, gap=gap)
    records: list[dict] = []

    X_arr = X.values
    y_arr = y.values
    dates = X.index

    model_factory = logistic_factory(class_weight="balanced", c_value=0.5)

    for fold_idx, (train_idx, test_idx) in enumerate(tscv.split(X_arr)):
        if len(train_idx) < min_train_obs:
            continue
        X_train, X_test = X_arr[train_idx], X_arr[test_idx]
        y_train, y_test = y_arr[train_idx], y_arr[test_idx]

        if len(np.unique(y_train)) < 2:
            continue

        # Impute NaN with training column medians
        col_medians = np.nanmedian(X_train, axis=0)
        for col_i in range(X_train.shape[1]):
            X_train[np.isnan(X_train[:, col_i]), col_i] = col_medians[col_i]
            X_test[np.isnan(X_test[:, col_i]), col_i] = col_medians[col_i]

        model = model_factory()
        model.fit(X_train, y_train)
        y_prob = model.predict_proba(X_test)[:, 1]

        for i, (true_val, prob_val) in enumerate(zip(y_test, y_prob)):
            records.append({
                "fold": fold_idx,
                "train_start": str(dates[train_idx[0]].date()),
                "train_end": str(dates[train_idx[-1]].date()),
                "test_date": str(dates[test_idx[i]].date()),
                "y_true": int(true_val),
                "y_prob": float(prob_val),
            })

    return pd.DataFrame(records)


# ---------------------------------------------------------------------------
# Metrics computation
# ---------------------------------------------------------------------------

def compute_metrics(results_df: pd.DataFrame, label: str) -> dict:
    """Compute pooled metrics over all WFO folds."""
    y_true = results_df["y_true"].values
    y_prob = results_df["y_prob"].values

    # Covered = outside abstention band
    covered_mask = (y_prob < ABSTAIN_LOW) | (y_prob > ABSTAIN_HIGH)
    coverage = float(covered_mask.mean())

    y_true_cov = y_true[covered_mask]
    y_prob_cov = y_prob[covered_mask]
    y_pred_cov = (y_prob_cov > 0.5).astype(int)

    return {
        "model": label,
        "n_obs": len(y_true),
        "n_covered": int(covered_mask.sum()),
        "coverage": round(coverage, 4),
        "balanced_accuracy_all": round(
            balanced_accuracy_score(y_true, (y_prob > 0.5).astype(int)), 4
        ),
        "balanced_accuracy_covered": (
            round(balanced_accuracy_score(y_true_cov, y_pred_cov), 4)
            if len(np.unique(y_true_cov)) > 1
            else None
        ),
        "brier_score": round(brier_score_loss(y_true, y_prob), 4),
        "log_loss": round(log_loss(y_true, y_prob), 4),
        "ece": round(_compute_ece(y_true, y_prob), 4),
        "base_rate_positive": round(float(y_true.mean()), 4),
    }
